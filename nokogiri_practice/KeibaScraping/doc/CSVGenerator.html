def generate_race_csv(race_data)
  CSV.open('race.csv', 'a+') do |csv|
    csv << [ race_data.place, race_data.num, race_data.name ]
    csv << [ '着順', '馬番', '名前', '騎手名', '人気', 'オッズ' ]
    race_data.result.each do |horce|
      csv << [ 
        horce.ranking,
        horce.number,
        horce.name,
        horce.jockey,
        horce.popularity,
        horce.oddsm
      ]
    end
  end
end
